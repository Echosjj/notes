发布订阅模式：

es6 观察者模式
class Observer{
  constructor(){
    this.handlers = []
  },
  addSubscribe(topic, fn){
    this.handlers.push({topic: topic, fn: fn})
  },
  removeSubscribe(topic, fn){
    this.handlers.forEach((item, index) => {
      if(item.topic == topic){
        this.handlers.splice(index, 1);
      }
    })
  },
  publish(topic, message){
    this.handlers.forEach((item, index) => {
      if(item.topic == topic){
        item.fn(message);
      }
    })
  }
}
export default new Observer();
